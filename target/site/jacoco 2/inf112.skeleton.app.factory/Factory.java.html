package inf112.skeleton.app.factory;

import java.util.HashMap;

public class Factory<T> {
    private final HashMap<String, Producer<T>> producers;
    private final HashMap<String, Object> args;

    private Factory(HashMap<String, Producer<T>> producers, HashMap<String, Object> args) {
        this.producers = producers;
        this.args = args;
    }

    public Factory() {
        this(new HashMap<>(), new HashMap<>());
    }

    public void addProducer(String name, Producer<T> producer) {
        this.producers.put(name, producer);
    }

    public void set(String argName, Object value) {
        args.put(argName, value);
    }

    public Object get(String argName) {
        return args.get(argName);
    }

    public T create(String name) {
        Producer<T> producer = producers.get(name);
        return producer.produce(args);
    }
}
