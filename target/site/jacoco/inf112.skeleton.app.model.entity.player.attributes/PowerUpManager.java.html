package inf112.skeleton.app.model.entity.player.attributes;

import com.badlogic.gdx.graphics.Texture;

import inf112.skeleton.app.view.Media;

public class PowerUpManager {
    private boolean godMode;
    private boolean rayGun;
    private int godModeTimer;
    private int rayGunTimer;
    private final int godModeDuration;
    private final int rayGunDuration;
    private Texture godModeTexture;

    public PowerUpManager(int godModeDuration, int rayGunDuration) {
        this.godModeDuration = godModeDuration;
        this.rayGunDuration = rayGunDuration;
        this.godMode = false;
        this.rayGun = false;
        this.godModeTexture = Media.player;
    }

    /**
     * Activates god mode for the player
     */

    public void activateGodMode() {
        godMode = true;
        godModeTimer = godModeDuration;
    }

    /**
     * Deactivates god mode for the player
     */

    public void deactivateGodMode() {
        godMode = false;
    }

    /**
     * Returns whether god mode is active or not
     * @return
     */

    public boolean isGodModeActive() {
        return godMode;
    }

    /**
     * Activates the ray gun for the player
     */

    public void activateRayGun() {
        rayGun = true;
        rayGunTimer = rayGunDuration;
    }

    /**
     *  Deactivates the ray gun for the player
     */

    public void deactivateRayGun() {
        rayGun = false;
    }

    /**
     * Returns whether the ray gun is active or not
     * @return
     */

    public boolean isRayGunActive() {
        return rayGun;
    }

    /**
     * Returns the texture of the player when god mode is active
     * @return
     */

    public Texture getGodModeTexture() {
        return godModeTexture;
    }
    
    /**
     * Updates the power up manager
     * @param deltaTime
     */

    public void update(float deltaTime) {
        if (godMode && godModeTimer > 0) {
            godModeTimer--;
            if (godModeTimer <= 0) {
                deactivateGodMode();
                this.godModeTexture = Media.player;
                return;
            }
            this.godModeTexture = Media.getGodModeColors()[godModeTimer % Media.getGodModeColors().length];
        }

        if (rayGun && rayGunTimer > 0) {
            rayGunTimer--;
            if (rayGunTimer <= 0) {
                deactivateRayGun();
            }
        }
    }
}
