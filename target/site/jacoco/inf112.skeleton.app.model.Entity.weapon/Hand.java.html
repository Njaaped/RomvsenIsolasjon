package inf112.skeleton.app.model.Entity.weapon;

import com.badlogic.gdx.math.Vector2;
import com.badlogic.gdx.physics.box2d.Body;

public class Hand implements Weapon, Cooldown {
  private Body parentBody;
  private int attackCooldown = 30;
  private int cooldownProgress = 0;
  private int damage = 5;
  private float attackDuration = 100;
  private float size = .3f;

  private Class<?> ownerType;

  public Hand(Body parentBody, Class<?> ownerType) {
    this.parentBody = parentBody;
    this.ownerType = ownerType;
  }

  @Override
  public void attack(Vector2 direction) {
    if (cooldownProgress < attackCooldown) {
      return;
    }

    Vector2 strikePosition = parentBody.getPosition().cpy();
    strikePosition.add(new Vector2(-size/2, -size/2).add(direction.scl(1f)));

    new Strike(parentBody, strikePosition, direction, size, damage, attackDuration, ownerType);
    cooldownProgress = 0;
  }

  @Override
  public void reduceCooldown() {
    if (cooldownProgress < attackCooldown) {
      cooldownProgress++;
    }
  }


}
